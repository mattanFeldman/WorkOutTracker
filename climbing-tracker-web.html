function storageConfigured(){

  return JSONBIN_API_KEY && JSONBIN_API_KEY.indexOf("PASTE_") !== 0

      && JSONBIN_BIN_ID && JSONBIN_BIN_ID.indexOf("PASTE_") !== 0;

}


async function loadState(){

  if(!storageConfigured()){

    showConfigBanner();

    state = { sessions: {} };

    return;

  }

  try{

    const res = await fetch(`https://api.jsonbin.io/v3/b/${JSONBIN_BIN_ID}/latest`, {

      headers: { "X-Master-Key": JSONBIN_API_KEY, "X-Bin-Meta": "false" }

    });

    if(!res.ok) throw new Error("fetch failed");

    const data = await res.json();

    state = (data && data.sessions) ? data : { sessions: {} };

  }catch(e){

    showToast("Couldn't load saved data — check connection/config");

    state = { sessions: {} };

  }

}


async function saveState(){

  if(!storageConfigured()){

    showConfigBanner();

    return;

  }

  try{

    const res = await fetch(`https://api.jsonbin.io/v3/b/${JSONBIN_BIN_ID}`, {

      method: "PUT",

      headers: { "Content-Type": "application/json", "X-Master-Key": JSONBIN_API_KEY },

      body: JSON.stringify(state)

    });

    if(!res.ok) throw new Error("save failed");

  }catch(e){

    showToast("Save failed — check connection/config");

  }

}

function getLastForExercise(exId, beforeDate){

  // find most recent session (strictly before or equal to `beforeDate`) containing this exercise

  const dates = Object.keys(state.sessions)

    .filter(d => state.sessions[d].entries && state.sessions[d].entries[exId])

    .sort()

    .reverse();

  for(const d of dates){

    if(d <= beforeDate){

      return { date:d, sets: state.sessions[d].entries[exId], mode: state.sessions[d].mode || 'climb' };

    }

  }

  return null;

}

      wrap.appendChild(tag);

      block.ids.forEach((exId, idx)=>{

        wrap.appendChild(buildExCard(findEx(exId), dateVal));

        if(idx < block.ids.length - 1){

          const divider = document.createElement('div');

          divider.className = 'superset-divider';

          divider.textContent = 'then';

          wrap.appendChild(divider);

        }

      });

      container.appendChild(wrap);

    } else {

      const wrap = document.createElement('div');

      wrap.className = 'solo-wrap';

      const tag = document.createElement('div');

      tag.className = 'group-tag';

      tag.innerHTML = `<span class="dot"></span> Standalone — full rest between sets`;

      wrap.appendChild(tag);

      wrap.appendChild(buildExCard(findEx(block.id), dateVal));

      container.appendChild(wrap);

    }

  });

}


function renderAll(){

  renderTabs();

  renderModeToggle();

  renderDayInfo();

  renderExercises();

}


async function logSession(){

  const dateVal = dateInput.value;

  const plan = PLAN[currentDay];

  const entries = {};

  let anyData = false;


  document.querySelectorAll('#exerciseList .ex-card').forEach(card=>{

    const exId = card.dataset.exid;

    const sets = [];

    const weightInputs = card.querySelectorAll('input[data-role="weight"]');

    const repInputs = card.querySelectorAll('input[data-role="reps"]');

    for(let i=0;i<weightInputs.length;i++){

      const w = weightInputs[i].value.trim();

      const r = repInputs[i].value.trim();

      if(w !== '' || r !== ''){

        sets.push({ w: w || '—', r: r || '—' });

        anyData = true;

      }

    }

    if(sets.length > 0){

      entries[exId] = sets;

    }

  });


  if(!anyData){

    showToast("Enter at least one weight or rep to log");

    return;

  }


  state.sessions[dateVal] = { day: currentDay, mode: currentMode, entries };

  await saveState();

  showToast(`Session logged for ${dateVal}`);

  renderExercises(); // refresh "last logged" hints

  renderHistoryIfOpen();

}


function renderHistoryIfOpen(){

  const panel = document.getElementById('historyPanel');

  if(panel.classList.contains('open')){

    renderHistory();

  }

}


function renderHistory(){

  const panel = document.getElementById('historyPanel');

  const plan = PLAN[currentDay];

  let html = '';

  plan.exercises.forEach(ex=>{

    const dates = Object.keys(state.sessions)

      .filter(d => state.sessions[d].entries && state.sessions[d].entries[ex.id])

      .sort().reverse().slice(0,5);

    if(dates.length === 0) return;

    html += `<div class="hist-ex-title">${ex.name}</div>`;

    dates.forEach(d=>{

      const sets = state.sessions[d].entries[ex.id];

      const tag = state.sessions[d].mode === 'rest' ? ' (rest)' : '';

      html += `<div class="hist-row"><span class="hd">${d}${tag}</span><span>${sets.map(s=>`${s.w}×${s.r}`).join(', ')}</span></div>`;

    });

  });

  panel.innerHTML = html || `<div class="last-hint">No sessions logged yet for Day ${currentDay}.</div>`;

}


document.getElementById('logBtn').addEventListener('click', logSession);
